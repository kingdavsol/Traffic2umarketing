Connect LinkedIn accounts to post on user's behalf:

**OAuth Flow:**
```typescript
// 1. Redirect to LinkedIn
const authUrl = `https://www.linkedin.com/oauth/v2/authorization?
  response_type=code&
  client_id=${LINKEDIN_CLIENT_ID}&
  redirect_uri=${REDIRECT_URI}&
  scope=w_member_social,r_basicprofile,r_organization_social
`;

// 2. Callback handler
// GET /api/linkedin/callback?code=...
async function handleCallback(code: string) {
  // Exchange code for access token
  const tokenResponse = await fetch(
    'https://www.linkedin.com/oauth/v2/accessToken',
    {
      method: 'POST',
      headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
      body: new URLSearchParams({
        grant_type: 'authorization_code',
        code,
        client_id: LINKEDIN_CLIENT_ID,
        client_secret: LINKEDIN_CLIENT_SECRET,
        redirect_uri: REDIRECT_URI,
      }),
    }
  );

  const { access_token } = await tokenResponse.json();

  // Get user profile
  const profile = await fetch('https://api.linkedin.com/v2/me', {
    headers: { Authorization: `Bearer ${access_token}` },
  }).then(r => r.json());

  // Store in database
  await prisma.linkedInAccount.create({
    data: {
      userId: currentUser.id,
      linkedinId: profile.id,
      accessToken: encrypt(access_token), // Encrypt!
      name: `${profile.localizedFirstName} ${profile.localizedLastName}`,
    },
  });
}
```

### 3. Post Scheduling System
**Priority:** CRITICAL
**Estimated Time:** 6-8 hours

Automatically publish posts at scheduled time:

**Cron Job:**
```typescript
// POST /api/cron/publish-posts
// Runs every 5 minutes

export async function POST(request: NextRequest) {
  // Get posts scheduled for now or past
  const postsToPublish = await prisma.post.findMany({
    where: {
      status: 'SCHEDULED',
      scheduledFor: {
        lte: new Date(),
      },
    },
    include: {
      linkedInAccount: true,
    },
  });

  // Publish each post
  for (const post of postsToPublish) {
    try {
      await publishToLinkedIn(post);

      // Update status
      await prisma.post.update({
        where: { id: post.id },
        data: {
          status: 'PUBLISHED',
          publishedAt: new Date(),
        },
      });
    } catch (error) {
      // Mark as failed
      await prisma.post.update({
        where: { id: post.id },
        data: { status: 'FAILED', error: error.message },
      });
    }
  }

  return NextResponse.json({ published: postsToPublish.length });
}
```

**LinkedIn Publishing:**
```typescript
// src/lib/linkedin/publisher.ts
async function publishToLinkedIn(post: Post & { linkedInAccount: LinkedInAccount }) {
  const accessToken = decrypt(post.linkedInAccount.accessToken);

  // Create post on LinkedIn
  const response = await fetch(
    'https://api.linkedin.com/v2/ugcPosts',
    {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${accessToken}`,
        'Content-Type': 'application/json',
        'X-Restli-Protocol-Version': '2.0.0',
      },
      body: JSON.stringify({
        author: `urn:li:person:${post.linkedInAccount.linkedinId}`,
        lifecycleState: 'PUBLISHED',
        specificContent: {
          'com.linkedin.ugc.ShareContent': {
            shareCommentary: { text: post.content },
            shareMediaCategory: post.images?.length > 0 ? 'IMAGE' : 'NONE',
            media: post.images?.map(url => ({
              status: 'READY',
              media: url,
            })),
          },
        },
        visibility: { 'com.linkedin.ugc.MemberNetworkVisibility': 'PUBLIC' },
      }),
    }
  );

  const data = await response.json();

  if (!response.ok) {
    throw new Error(data.message || 'Failed to publish');
  }

  // Store LinkedIn post ID
  await prisma.post.update({
    where: { id: post.id },
    data: { linkedinPostId: data.id },
  });

  return data;
}
```

```typescript
// src/lib/linkedin/analytics.ts
async function fetchPostAnalytics(
  accessToken: string,
  postId: string
) {
  const response = await fetch(
    `https://api.linkedin.com/v2/socialActions/${postId}`,
    {
      headers: {
        'Authorization': `Bearer ${accessToken}`,
        'X-Restli-Protocol-Version': '2.0.0',
      },
    }
  );

  const data = await response.json();

  return {
    likes: data.likesSummary?.totalLikes || 0,
    comments: data.commentsSummary?.totalComments || 0,
    shares: data.sharesSummary?.totalShares || 0,
  };
}

// Cron job to update analytics
// POST /api/cron/update-analytics
// Runs every hour

export async function POST() {
  // Get all published posts from last 30 days
  const posts = await prisma.post.findMany({
    where: {
      status: 'PUBLISHED',
      publishedAt: {
        gte: new Date(Date.now() - 30 * 24 * 60 * 60 * 1000),
      },
    },
    include: { linkedInAccount: true },
  });

  // Fetch analytics for each
  for (const post of posts) {
    const analytics = await fetchPostAnalytics(
      decrypt(post.linkedInAccount.accessToken),
      post.linkedinPostId
    );

    // Store in database
    await prisma.postAnalytics.upsert({
      where: { postId: post.id },
      create: { postId: post.id, ...analytics },
      update: analytics,
    });
  }

  return NextResponse.json({ updated: posts.length });
}
```
