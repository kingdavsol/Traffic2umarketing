```typescript
async function handleCallback(code: string, state: string) {
  // Exchange code for access token
  const response = await fetch(
    'https://connect.stripe.com/oauth/token',
    {
      method: 'POST',
      headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
      body: new URLSearchParams({
        grant_type: 'authorization_code',
        code,
        client_secret: STRIPE_SECRET_KEY,

  // Store in database
  await prisma.stripeAccount.create({
    data: {
      userId: decodeState(state).userId,
      stripeAccountId: stripe_user_id,
      accessToken: encrypt(access_token),
      refreshToken: encrypt(refresh_token),
    },
  });

  // Start initial sync
  await syncStripeData(stripe_user_id);
}
```

```typescript
// Initial sync (full historical data)
async function initialSync(stripeAccountId: string) {
  const account = await prisma.stripeAccount.findUnique({
    where: { stripeAccountId },
  });

  const stripe = new Stripe(decrypt(account.accessToken), {
    stripeAccount: stripeAccountId,
  });

  // 1. Sync customers
  for await (const customer of stripe.customers.list({ limit: 100 })) {
    await prisma.stripeCustomer.upsert({
      where: { stripeCustomerId: customer.id },
      create: {
        stripeAccountId,
        stripeCustomerId: customer.id,
        email: customer.email,
        name: customer.name,
        createdAt: new Date(customer.created * 1000),
      },
      update: {
        email: customer.email,
        name: customer.name,
      },
    });
  }

  // 2. Sync subscriptions
  for await (const subscription of stripe.subscriptions.list({ limit: 100 })) {
    await syncSubscription(stripeAccountId, subscription);
  }

  // 3. Sync charges (last 90 days)
  const charges = await stripe.charges.list({
    limit: 100,
    created: { gte: Math.floor(Date.now() / 1000) - 90 * 24 * 60 * 60 },
  });
  for (const charge of charges.data) {
    await syncCharge(stripeAccountId, charge);
  }

  // 4. Calculate MRR snapshots
  await calculateMRRSnapshots(stripeAccountId);
}

// Incremental sync (via webhooks)
async function handleWebhook(event: Stripe.Event, stripeAccountId: string) {
  switch (event.type) {
    case 'customer.created':
    case 'customer.updated':
      await syncCustomer(stripeAccountId, event.data.object);
      break;

    case 'customer.subscription.created':
    case 'customer.subscription.updated':
    case 'customer.subscription.deleted':
      await syncSubscription(stripeAccountId, event.data.object);
      await recalculateMRR(stripeAccountId);
      break;

    case 'charge.succeeded':
    case 'charge.refunded':
      await syncCharge(stripeAccountId, event.data.object);
      break;

    case 'invoice.paid':
    case 'invoice.payment_failed':
      await syncInvoice(stripeAccountId, event.data.object);
      break;
  }
}
```

### 3. MRR Calculation Engine
**Priority:** CRITICAL
**Estimated Time:** 8-10 hours

Calculate Monthly Recurring Revenue:

**MRR Formula:**
```typescript
MRR = Sum of (subscription amount normalized to monthly)

// Examples:
- $100/month subscription = $100 MRR
- $1200/year subscription = $100 MRR ($1200 / 12)
- $50/month × 10 customers = $500 MRR
```

**Implementation:**
```typescript
// src/lib/analytics/mrr.ts
async function calculateMRR(stripeAccountId: string, date: Date): Promise<number> {
  // Get all active subscriptions on this date
  const subscriptions = await prisma.stripeSubscription.findMany({
    where: {
      stripeAccountId,
      status: { in: ['active', 'trialing'] },
      createdAt: { lte: date },
      OR: [
        { canceledAt: null },
        { canceledAt: { gt: date } },
      ],
    },
  });

  let totalMRR = 0;

  for (const sub of subscriptions) {
    const amount = sub.amount;
    const interval = sub.interval; // 'month' or 'year'

    // Normalize to monthly
    const monthlyAmount = interval === 'year'
      ? amount / 12
      : amount;

    totalMRR += monthlyAmount;
  }

  return totalMRR / 100; // Convert from cents to dollars
}

// Calculate and store daily snapshots
async function calculateMRRSnapshots(stripeAccountId: string) {
  const startDate = new Date('2024-01-01'); // Or account creation date
  const endDate = new Date();

  // Calculate MRR for each day
  for (let date = startDate; date <= endDate; date.setDate(date.getDate() + 1)) {
    const mrr = await calculateMRR(stripeAccountId, new Date(date));

    await prisma.mRRSnapshot.upsert({
      where: {
        stripeAccountId_date: {
          stripeAccountId,
          date: new Date(date),
        },
      },
      create: {
        stripeAccountId,
        date: new Date(date),
        mrr,
      },
      update: { mrr },
    });
  }
}
```
